Recursion Simulator Code


def recursion_simulator(function, *args, depth=0):
    """
    Simulates recursion by printing the call stack depth and arguments.
    """
    print(f"{'  ' * depth}Recursion depth {depth}: {function.__name__}({', '.join(map(str, args))})")
    result = function(*args, depth=depth + 1)
    print(f"{'  ' * depth}Returning from depth {depth}: {result}")
    return result

# Example recursive function
def factorial(n, depth=0):
    if n == 0:
        return 1
    return n * recursion_simulator(factorial, n - 1, depth=depth)

# Test the recursion simulator
if __name__ == "__main__":
    print("Factorial result:", recursion_simulator(factorial, 5))
